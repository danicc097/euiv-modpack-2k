# El Cid appears in Castile
country_event = {
    id = cid_events.1
    title = cid_events.1.t
    desc = cid_events.1.d
    picture = CONQUEST_eventPicture

    fire_only_once = yes

    trigger = {
        NOT = { is_year = 1075 }
        is_year = 1054
        normal_or_historical_nations = yes
        tag = CAS
        government = monarchy
        religion_group = christian
    }
    
    mean_time_to_happen = {
        months = 48
        modifier = {
            factor = 0.5
            is_year = 1060
        }
    }

    option = {
        name = cid_events.1.a
        ai_chance = {
            factor = 65
            modifier = {
                factor = 2
                NOT = { num_of_generals = 1 }
            }
            modifier = {
                factor = 0
                num_of_generals = 3
            }
        }
        define_general = {
            name = "El Cid"
            fire = 4
            shock = 5
            manuever = 4
            siege = 2
        }
        set_country_flag = cid_cas
    }
    option = {
        name = cid_events.1.b
        ai_chance = {
            factor = 35
            modifier = {
                factor = 2
                num_of_generals = 3
            }
            modifier = {
                factor = 0
                NOT = { num_of_generals = 1 }
            }
        }
        custom_tooltip = cid_events.1.tt
        hidden_effect = {
            if = {
                limit = { exists = BRC }
                BRC = {
                    country_event = {
                        id = cid_events.2
                        days = 60
                        random = 5
                    }
                }
            }
            else = {
                if = {
                    limit = { exists = ZGZ }
                    ZGZ = {
                        country_event = {
                            id = cid_events.3
                            days = 40
                            random = 5
                        }
                    }
                }
                else = {
                    random_country = {
                        limit = {
                            capital_scope = {
                                superregion = iberia_superregion
                            }
                            religion_group = muslim
                        }
                        country_event = {
                            id = cid_events.4
                            days = 50
                            random = 5
                        }
                    }
                }
            }
        }
    }
}

            factor = 0.5
            is_year = 1080

            factor = 0.5
            is_year = 1090
        }
    }

    immediate = {
        hidden_effect = {
            if = {
                limit = { has_country_flag = cid_cas }
                clr_country_flag = cid_cas

                limit = { has_country_flag = cid_brc }

    }

    option = {
        name = cid_events.5.a

            2024 = {
                NOT = { country_or_non_sovereign_subject_holds = ROOT }

            kill_leader = {
                type = "El Cid"
            }

                owner = {
                    country_event = {
                        id = cid_events.6
                        days = 30
                        random = 10
                    }
                }
            }
        }
        custom_tooltip = cid_events.5.tt
    }
    option = {
        name = cid_events.5.b

            2024 = {
                NOT = { country_or_non_sovereign_subject_holds = ROOT }
            }
        }
        ai_chance = { 
            factor = 30

                factor = 0.5
                treasury = 100
            }

                factor = 0.5
                manpower = 2
            }

# El Cid's Campaign
country_event = {
    id = cid_events.6
    title = cid_events.6.t
    desc = cid_events.6.d
    picture = SIEGE_eventPicture

    is_triggered_only = yes

    immediate = {
        hidden_effect = {
            every_owned_province = {
                limit = { area = valencia_area }
                add_core = AVL
                cede_province = AVL
            }
            AVL = {
                define_ruler = {
                    name = "Rodrigo Díaz"
                    dynasty = "Laínez"
                    min_age = 40
                    max_age = 55
                    culture = castillian
                    adm = 3
                    dip = 5
                    mil = 6
                }
                define_ruler_to_general = {
                    fire = 4
                    shock = 5
                    manuever = 4
                    siege = 2
                }
                add_treasury = 100
                add_manpower = 2
            }
            set_country_flag = cid_rebelled  
        }
    }

    option = {
        name = cid_events.6.a
        ai_chance = { 
            factor = 70
            modifier = {
                factor = 0.5
                is_at_war = yes
            }
        }
        AVL = {
            declare_war_with_cb = {
                casus_belli = cb_independence_war
                who = ROOT
            }
        }
        hidden_effect = {
            if = {
                limit = {
                    FROM = { has_country_flag = cid_sent_help }
                }
                FROM = {
                    create_alliance = AVL
                    clr_country_flag = cid_sent_help
                }
            }
        }
        custom_tooltip = cid_events.6.a.tt
    }
    option = {
        name = cid_events.6.b
        ai_chance = {
            factor = 30
            modifier = {
                factor = 2.0
                is_at_war = yes
            }
            modifier = {
                factor = 1.5
                NOT = { manpower = 1 }
            }
            modifier = {
                factor = 1.2
                num_of_loans = 1
            }
        }
        AVL = {
            white_peace = ROOT
            add_truce_with = ROOT
        }
        hidden_effect = {
            clr_country_flag = cid_rebelled
            if = {
                limit = {
                    FROM = { has_country_flag = cid_sent_help }
                }
                FROM = {
                    create_alliance = AVL
                    clr_country_flag = cid_sent_help
                }
            }
        }
    }
}

# El Cid is Victorious
country_event = {
    id = cid_events.7
    title = cid_events.7.t
    desc = cid_events.7.d
    picture = SIEGE_eventPicture

    fire_only_once = yes

    trigger = {
        2024 = {
            owned_by = AVL
            controlled_by = AVL
        }
        has_country_flag = cid_rebelled
        had_country_flag = {
            flag = cid_rebelled
            days = 365
        }
    }

    mean_time_to_happen = {
        months = 1
    }

    option = {
        name = cid_events.7.a
        ai_chance = { factor = 100 }
        hidden_effect = {
            clr_country_flag = cid_rebelled
        }
        valencia_area = {
            limit = { owned_by = ROOT }
            add_core = AVL
            cede_province = AVL
        }
        AVL = {
            every_known_country = {
                limit = { war_with = AVL }
                white_peace = AVL
                add_truce_with = AVL
            }
        }
    }
}

# El Cid is Defeated
country_event = {
    id = cid_events.8
    title = cid_events.8.t
    desc = cid_events.8.d
    picture = WOUNDED_SOLDIERS_eventPicture

    fire_only_once = yes

    trigger = {
        has_country_flag = cid_rebelled
        had_country_flag = {
            flag = cid_rebelled
            days = 365
        }
        OR = {
            AND = {
                war_with = AVL
                2024 = {
                    NOT = { controlled_by = AVL }
                }
            }
            2024 = {
                NOT = { owned_by = AVL }
            }
        }
    }

    mean_time_to_happen = {
        months = 1
    }

    option = {
        name = cid_events.8.a
        ai_chance = { factor = 100 }
        clr_country_flag = cid_rebelled
        add_prestige = 10
        2024 = {
            add_devastation = 50
            add_province_modifier = {
                name = "cid_valencia_devastated"
                duration = 1095
            }
        }
        if = {
            limit = { exists = AVL }
            AVL = {
                every_owned_province = {
                    cede_province = ROOT
                    remove_core = AVL
                }
            }
        }
        custom_tooltip = cid_events.8.a.tt
    }
}
