# Local Governor enacts trade trade laws
country_event = {
	id = china_events.1
	title = "china_events.1.t"
	desc = "china_events.1.d"
	picture = ADVISOR_eventPicture
	
	trigger = {
		any_owned_province = {
			is_capital = no
			is_state = yes
			NOT = { has_province_modifier = chi_local_trade_regulations }
			NOT = { has_province_modifier = chi_local_trade_down_turn }
		}
		OR = {
			AND = {
				NOT = { has_dlc = "Mandate of Heaven" }
				has_reform = celestial_empire
			}
			AND = {
				has_dlc = "Mandate of Heaven"
				is_emperor_of_china = yes
			}
		}
	}
	
	is_triggered_only = yes
	
	immediate = {
		hidden_effect = {
			random_owned_province = { #Atleast here
				limit = {
					is_capital = no
					is_state = yes
					NOT = { has_province_modifier = chi_local_trade_regulations }
					NOT = { has_province_modifier = chi_local_trade_down_turn }
				}
				save_event_target_as = trade_problems_province
			}
			random_owned_province = { #But rather here
				limit = {
					is_capital = no
					has_building = marketplace
					NOT = { has_province_modifier = chi_local_trade_regulations }
					NOT = { has_province_modifier = chi_local_trade_down_turn }
				}
				save_event_target_as = trade_problems_province
			}
			random_owned_province = { #Here would be even better
				limit = {
					is_capital = no
					has_port = yes
					NOT = { has_province_modifier = chi_local_trade_regulations }
					NOT = { has_province_modifier = chi_local_trade_down_turn }
				}
				save_event_target_as = trade_problems_province
			}
			random_owned_province = { #Warmer
				limit = {
					is_capital = no
					has_port = yes
					has_building = marketplace
					NOT = { has_province_modifier = chi_local_trade_regulations }
					NOT = { has_province_modifier = chi_local_trade_down_turn }
				}
				save_event_target_as = trade_problems_province
			}
			random_owned_province = { #Best
				limit = {
					is_capital = no
					province_has_center_of_trade_of_level = 1
					NOT = { has_province_modifier = chi_local_trade_regulations }
					NOT = { has_province_modifier = chi_local_trade_down_turn }
				}
				save_event_target_as = trade_problems_province
			}
		}
	}
	
	mean_time_to_happen = {
		days = 1
	}
	
	option = {
		name = "china_events.1.a"
		reduce_mandate_effect = yes
		if = {
			limit = {
				NOT = { has_dlc = "Mandate of Heaven" }
			}
			add_legitimacy = -5
		}
		event_target:trade_problems_province = {
			add_province_modifier = {
				name = "chi_local_trade_regulations"
				duration = 3650
			}
		}
	}
	
	option = {
		name = "china_events.1.b"
		add_mandate_effect = yes
		event_target:trade_problems_province = {
			add_province_modifier = {
				name = "chi_local_trade_down_turn"
				duration = 3650
			}
		}
	}
}

country_event = {
	id = china_events.14
	title = "china_events.14.t"
	desc = "china_events.14.d"
	picture = MILITARY_CAMP_eventPicture
	
	is_triggered_only = yes
	
	trigger = {
		OR = {
			AND = {
				NOT = { has_dlc = "Mandate of Heaven" }
				has_reform = celestial_empire
			}
			AND = {
				has_dlc = "Mandate of Heaven"
				is_emperor_of_china = yes
			}
			has_country_modifier = lost_mandate_of_heaven
		}
		any_owned_province = {
			fort_level = 2
		}
		has_leaders = {
			value = 1
			type = general
			include_monarch = no
			include_heir = no
		}
	}
	
	immediate = {
		hidden_effect = {
			random_owned_province = {
				limit = {
					fort_level = 2
				}
				save_event_target_as = corrupt_fort
				set_province_flag = corrupt_fort_flag
			}
			random_owned_province = {
				limit = {
					fort_level = 2
					NOT = { has_province_flag = corrupt_fort_flag }
				}
				save_event_target_as = corrupt_fort2
			}
		}
	}
	
	mean_time_to_happen = {
		days = 1
	}
	
	option = {
		name = "china_events.14.a"
		event_target:corrupt_fort = {
			add_province_modifier = {
				name = "chi_forts_underfunded"
				duration = 3650
			}
			clr_province_flag = corrupt_fort_flag
		}
		event_target:corrupt_fort2 = {
			add_province_modifier = {
				name = "chi_forts_underfunded"
				duration = 3650
			}
		}
		kill_leader = {
			type = general
		}
	}
	
	option = {
		name = "china_events.14.b"
		reduce_mandate_effect = yes
		add_mil_power = -50
		if = {
			limit = {
				OR = {
					NOT = { has_dlc = "Mandate of Heaven" }
					has_country_modifier = lost_mandate_of_heaven
				}
			}
			add_legitimacy = -5
		}
	}
}

# 'When the people have enough, how can the ruler alone have too little?'
# Based on Wang Ao's 8-legged essay.
country_event = {
	id = china_events.16
	title = "china_events.16.t"
	desc = "china_events.16.d"
	picture = UNIVERSITY_eventPicture
	
	is_triggered_only = yes
	
	trigger = {
		NOT = { has_country_modifier = chi_efficient_taxation }
		OR = {
			AND = {
				NOT = { has_dlc = "Mandate of Heaven" }
				has_reform = celestial_empire
			}
			AND = {
				has_dlc = "Mandate of Heaven"
				is_emperor_of_china = yes
				meritocracy = 70
			}
		}
		stability = 1
		any_owned_province = {
			is_state = yes
			NOT = { devastation = 1 }
			NOT = { has_province_modifier = chi_lowered_taxes_in_state }
		}
	}
	
	immediate = {
		hidden_effect = {
			random_owned_province = {
				limit = {
					is_state = yes
					NOT = { devastation = 1 }
					NOT = { has_province_modifier = chi_lowered_taxes_in_state }
				}
				save_event_target_as = prosperity_boost_here
			}
		}
	}
	
	mean_time_to_happen = {
		days = 1
	}
	
	option = {
		name = "china_events.16.a"
		goto = prosperity_boost_here
		if = {
			limit = { has_dlc = "Mandate of Heaven" }
			event_target:prosperity_boost_here = {
				if = {
					limit = { NOT = { prosperity = 100 } }
					add_prosperity = 25
				}
				area = {
					add_province_modifier = {
						name = "chi_lowered_taxes_in_state"
						duration = 1825
					}
				}
			}
		}
		if = {
			limit = { NOT = { has_dlc = "Mandate of Heaven" } }
			add_stability_or_adm_power = yes
		}
	}
	
	option = {
		name = "china_events.16.b"
		reduce_meritocracy_effect = yes
		add_country_modifier = {
			name = "chi_efficient_taxation"
			duration = 3650
		}
	}
}
