country_event = {
	id = byz_crisis.8
	title = "byz_crisis.8.t"
	desc = "byz_crisis.8.d"
	picture = BAD_WITH_MONARCH_eventPicture

	fire_only_once = yes
	is_triggered_only = yes
	trigger = {
		NOT = { exists = AHL }
		any_owned_province = {
			region = corduene_region
			NOT = { area = taron_area }
			NOT = { area = sasun_area }
		}
	}
	mean_time_to_happen = { days = 1 }

	immediate = {
		hidden_effect = {
			corduene_region = {
				limit = {
					owned_by = ROOT
					NOT = { area = taron_area }
					NOT = { area = sasun_area }
				}
				add_core = AHL
			}
			release = AHL
			if = {
				limit = { AHL = { ai = yes } }
				AHL = {
					change_government = monarchy
					add_government_reform = iqta
					if = {
						limit = { is_part_of_hre = yes }
						set_in_empire = no
					}
					set_government_rank = 3
					add_stability = 3
					add_accepted_culture = armenian
					define_ruler = {
						name = "Sökmen"
						dynasty = "Sökmenoglu"
					}
					random_owned_province = {
						build_to_forcelimit = {
							infantry = 0.7
							cavalry = 0.3
						}
					}
				}
			}	
		}
	}

	option = {
		name = "byz_crisis.8.a"
		ai_chance = { factor = 100 }
		custom_tooltip = byz_crisis.tt
		hidden_effect = {
			AHL = {
				every_owned_province = {
					limit = { is_core = BYZ }
					remove_core = BYZ
					if = {
						limit = { religion = orthodox }
						change_religion = sunni
					}
					if = {
						limit = { culture_group = byzantine }
						change_culture = turkish
					}
				}
			}
		}
	}
	option = {
		name = "byz_crisis.8.b"
		ai_chance = { factor = 0 }
		declare_war_with_cb = {
			who = AHL
			casus_belli = cb_disloyal_vassal
		}
	}
}
