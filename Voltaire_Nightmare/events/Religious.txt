# Heretic Conversion
province_event = {
	id = 6163
	title = "EVTNAME6163"
	desc = "EVTDESC6163"
	picture = PRAYING_eventPicture

	trigger = {
		is_capital = no
		owner = {
			NOT = { has_reform = papacy_reform }
			NOT = { religion_group = pagan }
			num_of_religion = { heretic = yes value = 2 }
		}  
		any_neighbor_province = {
			NOT = { religion = ROOT }
			religion_group = ROOT
		}
		NOT = { has_province_modifier = religious_zeal_at_conv }
	}	

	mean_time_to_happen = {
		months = 5000
		modifier = {
			factor = 0.95
			owner = { innovativeness_ideas = 4 }
		}
		modifier = {
			factor = 0.95
			owner = { innovativeness_ideas = 5 }
		}
		modifier = {
			factor = 0.95
			owner = { innovativeness_ideas = 6 }
		}
		modifier = {
			factor = 0.95
			owner = { innovativeness_ideas = 7 }
		}
		modifier = {
			factor = 2.0
			owner = { government = theocracy }
		}
	}

	is_mtth_scaled_to_size = 1

	option = {
		name = "EVTOPTA6163"
		change_religion = heretic
	}
}
