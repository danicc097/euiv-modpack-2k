# Portuguese Missionaries
country_event = {
	id = flavor_kon.1
	title = flavor_kon.1.t
	desc = flavor_kon.1.d
	picture = RELIGION_eventPicture
	fire_only_once = yes

	trigger = {
		tag = LIT
		POL = {
			OR = {
				religion = catholic
				religion = catholique
			}
		}
		knows_country = POL
		POL = { knows_country = ROOT }
		religion_group = pagan
		is_year = 1300
		NOT = { is_year = 1500 }
		#NOT = { has_dlc = "Origins" }
	}

	mean_time_to_happen = {
		months = 120
	}
	
	
	option = { 
		name = flavor_kon.1.a			# Let Them Preach
		ai_chance = { factor = 90 }
		
		add_stability = -1
		add_prestige = -25
		add_legitimacy = -25
		if = {
			limit = { 
				POL = { religion = catholic }
			}
			random_owned_province = {
				limit = {
					NOT = { religion = catholic }
				}
				change_religion = catholic
			 }
		}
		else = {
			random_owned_province = {
				limit = {
					NOT = { religion = catholique }
				}
				change_religion = catholique
			 }
		}
		set_country_flag = kongo_allows_missionaries
		
		reverse_add_opinion = {
			who = POL
			modifier = missionaries_accepted
		}

		hidden_effect = {
			POL = {
				country_event = { id = flavor_kon.2 days = 1 }
			}
		}
	}
	option = {
		name = flavor_kon.1.b			# Drive off the missionaries!
		ai_chance = { factor = 10 }
		
		add_stability_or_adm_power = yes
		add_prestige = 10
		
		reverse_add_opinion = {
			who = POL
			modifier = missionaries_rebuffed
		}
		
		hidden_effect = {
			POL = {
				country_event = { id = flavor_kon.3 days = 1 }
			}		
		}
	}
	
	
}
