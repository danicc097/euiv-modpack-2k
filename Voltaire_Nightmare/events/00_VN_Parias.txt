#Events for the paying of tribute by Muslim taifas to Christian kings

namespace = parias_events

#tribute
country_event = {
    id = parias_events.1
    title = "parias_events.1.t"
    desc = "parias_events.1.d"
    picture = LIBERUM_VETO_eventPicture

    is_triggered_only = yes

    trigger = {
        has_country_modifier = parias_modifier
        if = {
            limit = {
                OR = {
                    tag = ZGZ
                    tag = LER
                    tag = TTO
                }
            }
            OR = {
                AND = {
                    NOT = { is_in_war = { participants = BRC } }
                    exists = BRC
                    BRC = { is_subject = no }
                }
                AND = {
                    NOT = { is_in_war = { participants = CAT } }
                    exists = CAT
                    CAT = { is_subject = no }
                }
                AND = {
                    NOT = { is_in_war = { participants = ARA } }
                    exists = ARA
                    ARA = { is_subject = no }
                }
                AND = {
                    NOT = { is_in_war = { participants = NAV } }
                    exists = NAV
                    NAV = { is_subject = no }
                }
            }
        }
        else_if = {
            limit = {
                OR = {
                    tag = BDJ
                    tag = STN
                    tag = BJA
                    tag = SIV
                    tag = SEV
                    tag = AGB
                }
            }
            OR = {
                AND = {
                    NOT = { is_in_war = { participants = LON } }
                    exists = LON
                    LON = { is_subject = no }
                }
                AND = {
                    NOT = { is_in_war = { participants = GAL } }
                    exists = GAL
                    GAL = { is_subject = no }
                }
            }
        }
        else_if = {
            limit = {
                OR = {
                    tag = CTY
                    tag = MLN
                    tag = ARR
                    tag = ALP
                    tag = JRC
                    tag = VLK
                }
            }
            OR = {
                AND = {
                    NOT = { is_in_war = { participants = LON } }
                    exists = LON
                    LON = { is_subject = no }
                }
                AND = {
                    NOT = { is_in_war = { participants = CAS } }
                    exists = CAS
                    CAS = { is_subject = no }
                }
            }
        }
        else = {
            NOT = { is_in_war = { participants = LON } }
            exists = LON
            LON = { is_subject = no }
        }
    }

    immediate = {
        hidden_effect = {
            export_to_variable = {
                which = paria_amount
                value = monthly_income
                who = ROOT
            }
            multiply_variable = {
                which = paria_amount
                value = 3
            }
            export_to_variable = {
                which = paria_amount_halved
                value = monthly_income
                who = ROOT
            }
            multiply_variable = {
                which = paria_amount_halved
                value = 1.5
            }
            #select receiver
            if = {
                limit = {
                    OR = {
                        tag = ZGZ
                        tag = LER
                        tag = TTO
                    }
                }
                if = {
                    limit = {
                        NOT = { is_in_war = { participants = CAT } }
                        exists = CAT
                        CAT = { is_subject = no }
                    }
                    CAT = {
                        save_event_target_as = paria_receiver
                    }
                }
                else_if = {
                    limit = {
                        NOT = { is_in_war = { participants = BRC } }
                        exists = BRC
                        BRC = { is_subject = no }
                    }
                    BRC = {
                        save_event_target_as = paria_receiver
                    }
                }
                else_if = {
                    limit = {
                        NOT = { is_in_war = { participants = NAV } }
                        exists = NAV
                        NAV = { is_subject = no }
                    }
                    NAV = {
                        save_event_target_as = paria_receiver
                    }
                }
                else = {
                    ARA = {
                        save_event_target_as = paria_receiver
                    }
                }
            }
            else_if = {
                limit = {
                    OR = {
                        tag = BDJ
                        tag = STN
                        tag = BJA
                        tag = SIV
                        tag = SEV
                        tag = AGB
                    }
                }
                if = {
                    limit = {
                        NOT = { is_in_war = { participants = GAL } }
                        exists = GAL
                        GAL = { is_subject = no }
                    }
                    GAL = {
                        save_event_target_as = paria_receiver
                    }
                }
                else = {
                    LON = {
                        save_event_target_as = paria_receiver
                    }
                }
            }
            else_if = {
                limit = {
                    OR = {
                        tag = CTY
                        tag = MLN
                        tag = ARR
                        tag = ALP
                        tag = JRC
                        tag = VLK
                    }
                }
                if = {
                    limit = {
                        NOT = { is_in_war = { participants = CAS } }
                        exists = CAS
                        CAS = { is_subject = no }
                    }
                    CAS = {
                        save_event_target_as = paria_receiver
                    }
                }
                else = {
                    LON = {
                        save_event_target_as = paria_receiver
                    }
                }
            }
            else = {
                LON = {
                    save_event_target_as = paria_receiver
                }
            }

        }
    }

    option = {
        name = "parias_events.1.a"
        ai_chance = {
            factor = 20
            modifier = {
                factor = 0
                OR = {
                    num_of_loans = 1
                    is_bankrupt = yes
                    is_in_deficit = yes
                }
            }
        }
        custom_tooltip = parias_events.1.a.tt
        hidden_effect = {
            event_target:paria_receiver = {
                country_event = {
                    id = parias_events.6
                    days = 1
                }
            }
        }
    }
    option = {
        name = "parias_events.1.b"
        ai_chance = {
            factor = 60
            modifier = {
                factor = 0.5
                num_of_rebel_armies = 1
            }
            modifier = {
                factor = 0
                num_of_rebel_armies = 3
            }
            modifier = {
                factor  = 1.5
                NOT = { unrest = 0 }
            }
        }
        custom_tooltip = parias_events.1.b.tt
        if = {
            limit = { has_country_modifier = parias_taxes }
            extend_country_modifier = {
                name = parias_taxes
                duration = 730
            }
        }
        else = {
            add_country_modifier = {
                name = parias_taxes
                duration = 730
            }
        }
        hidden_effect = {
            event_target:paria_receiver = {
                country_event = {
                    id = parias_events.7
                    days = 1
                }
            }
        }
    }
    option = {
        name = "parias_events.1.c"
        trigger = {
            NOT = { has_country_modifier = parias_restored }
        }
        ai_chance = {
            factor = 10
            modifier = {
                factor = 0
                OR = {
                    num_of_loans = 1
                    is_bankrupt = yes
                    is_in_deficit = yes
                    is_at_war = yes
                    NOT = { is_year = 1060 }
                }
            }
            modifier = {
                factor = 0.1
                NOT = {
                    military_strength = {
                        who = event_target:paria_receiver
                        value = 1
                    }
                }
            }
            modifier = {
                factor = 1.5
                military_strength = {
                    who = event_target:paria_receiver
                    value = 1.5
                }
            }
            modifier = {
                factor = 1.1
                AND = {
                    army_size = 10
                    army_size_percentage = 0.9
                }
            }
            modifier = {
                factor = 1.1
                AND = {
                    manpower = 10
                    manpower_percentage = 0.9
                }
            }
            modifier = {
                factor = 1.1
                num_of_generals = 2
            }
            modifier = {
                factor = 1.1
                ruler_has_personality = conqueror_personality
            }
            modifier = {
                factor = 0.9
                ruler_has_personality = craven_personality
            }
            modifier = {
                factor = 0.2
                NOT = { manpower_percentage = 0.5 }
            }
            modifier = {
                factor = 0.2
                war_exhaustion = 1
            }
            modifier = {
                factor = 0.8
                NOT = { personality = ai_militarist }
            }
            modifier = {
                factor = 0.8
                NOT = { stability = 1 }
            }
        }
        remove_country_modifier = parias_modifier
        custom_tooltip = parias_events.1.c.tt
        hidden_effect = {
            event_target:paria_receiver = {
                country_event = {
                    id = parias_events.2
                    days = 1
                }
            }
        }
    }
}

                    id = parias_events.4

                }

        }
