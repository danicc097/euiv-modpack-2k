ate_give_advisor_based_on_gov_rank = {
	if = {
		limit = {
			OR = {
				government_rank = 3
				AND = {
					NOT = { has_dlc = "Common Sense" }
					total_development = 1000
				}
			}
		}
		define_advisor = {
			name = $name$
			type = $type$
			skill = 3
			culture = ROOT
			religion = ROOT
			cost_multiplier = 0.75
			female = no
			[[power_type]
			location = event_target:ate_possible_random_advisor_$power_type$_location
			]
		}
	}
	else_if = {
		limit = {
			OR = {
				government_rank = 2
				AND = {
					NOT = { has_dlc = "Common Sense" }
					total_development = 300
				}
			}
		}
		define_advisor = {
			name = $name$
			type = $type$
			skill = 2
			culture = ROOT
			religion = ROOT
			cost_multiplier = 0.75
			female = no
			[[power_type]
			location = event_target:ate_possible_random_advisor_$power_type$_location
			]
		}
	}
	else = {
		define_advisor = {
			name = $name$
			type = $type$
			skill = 1
			culture = ROOT
			religion = ROOT
			cost_multiplier = 0.75
			female = no
			[[power_type]
			location = event_target:ate_possible_random_advisor_$power_type$_location
			]
		}
	}
}
